import 'package:flutter/material.dart';
import 'package:flutter_test/flutter_test.dart';
import 'package:mobile_project_app/login_page.dart';

void main() {
  testWidgets('LoginPage basic UI test', (WidgetTester tester) async {
    // Запускаем LoginPage
    await tester.pumpWidget(
      MaterialApp(
        home: LoginPage(),
      ),
    );

    // Проверяем, что основные виджеты отображаются
    expect(find.text('Добро пожаловать в Timely'), findsOneWidget);
    expect(find.byType(TextField), findsNWidgets(2)); // email и пароль
    expect(find.text('Войти'), findsOneWidget);
    expect(find.text('Нет аккаунта? Зарегистрируйтесь'), findsOneWidget);
  });
}
